这篇博客来看一下构造函数、拷贝构造函数、赋值运算函数的区别和调用时机

```c++
class Test{
public:
    //无参构造函数
    Test(){
        printf("Enter ctr!\n");
    }
    //拷贝构造函数
    Test(const Test& other){
        printf("Enter copy!\n");
    }
    //赋值运算函数
    Test& operator= (const Test& other){
        printf("Enter assign!\n");
    }
    ~A(){
        printf("Enter dtr\n");
    }

}

int main(){
    A a;//调用构造函数
    A b = a;//调用拷贝构造函数，划重点，这里调用的是拷贝构造函数
    a = b;//调用赋值运算函数
}



```
